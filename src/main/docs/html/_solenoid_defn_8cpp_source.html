45 shared_ptr<DragonSolenoid> SolenoidDefn::ParseXML
46 (
47  xml_node solenoidNode
48 )
49 {
50  // initialize the output
51  shared_ptr<DragonSolenoid> solenoid;
52 
53  // initialize attributes to default values
54  SolenoidUsage::SOLENOID_USAGE usage = SolenoidUsage::SOLENOID_USAGE::UNKNOWN_SOLENOID_USAGE;
55  int pcmID = 0;
56  int channel = 0;
57  bool reversed = false;
58 
59  bool hasError = false;
60 
61  for (xml_attribute attr = solenoidNode.first_attribute(); attr && !hasError; attr = attr.next_attribute())
62  {
63  if ( strcmp( attr.name(), "usage" ) == 0 )
64  {
65  usage = SolenoidUsage::GetInstance()->GetUsage( string(attr.value()));
66  }
67  else if ( strcmp( attr.name(), "canId" ) == 0 )
68  {
69  pcmID = attr.as_int();
70  hasError = HardwareIDValidation::ValidateCANID( pcmID, string( "SolenoidDefn::ParseXML" ) );
71  }
72  else if ( strcmp( attr.name(), "channel" ) == 0 )
73  {
74  channel = attr.as_int();
75  hasError = HardwareIDValidation::ValidateSolenoidChannel( channel, string( "SolenoidDefn::ParseXML" ) );
76  }
77  // brake mode (or coast)
78  else if ( strcmp( attr.name(), "reversed" ) == 0 )
79  {
80  reversed = attr.as_bool();
81  }
82  else
83  {
84  string msg = "Invalid attribute";
85  msg += attr.name();
86  Logger::GetLogger()->LogError( string("Solenoid::ParseXML "), msg );
87  hasError = true;
88  }
89  }
90 
91  if ( !hasError )
92  {
93  solenoid = make_shared<DragonSolenoid>( usage, pcmID, channel, reversed );

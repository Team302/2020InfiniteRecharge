48 shared_ptr<IDragonMotorController> MotorDefn::ParseXML
49 (
50  xml_node motorNode
51 )
52 {
53  // initialize the output
54  shared_ptr<IDragonMotorController> controller;
55 
56  // initialize attributes to default values
57  int canID = 0;
58  int pdpID = -1;
59  string usage;
60  bool inverted = false;
61  bool sensorInverted = false;
62  ctre::phoenix::motorcontrol::FeedbackDevice feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::QuadEncoder;
63  int countsPerRev = 0;
64  float gearRatio = 1;
65  bool brakeMode = false;
66  int slaveTo = -1;
67  int peakCurrentDuration = 0;
68  int continuousCurrentLimit = 0;
69  int peakCurrentLimit = 0;
70  bool enableCurrentLimit = false;
71  bool forwardLimitSwitch = false;
72  bool forwardLimitSwitchNormallyOpen = false;
73  bool reverseLimitSwitch = false;
74  bool reverseLimitSwitchNormallyOpen = false;
75 
76 
77  string mtype;
78 
79 
80  bool hasError = false;
81 
82  for (xml_attribute attr = motorNode.first_attribute(); attr && !hasError; attr = attr.next_attribute())
83  {
84  if ( strcmp( attr.name(), "usage" ) == 0 )
85  {
86  usage = attr.value();
87  }
88  // CAN ID 0 thru 62 are valid
89  else if ( strcmp( attr.name(), "canId" ) == 0 )
90  {
91  canID = attr.as_int();
92  hasError = HardwareIDValidation::ValidateCANID( canID, string( "MotorDefn::ParseXML" ) );
93  }
94  // PDP ID 0 thru 15 are valid
95  else if ( strcmp( attr.name(), "pdpID" ) == 0 )
96  {
97  pdpID = attr.as_int();
98  hasError = HardwareIDValidation::ValidatePDPID( pdpID, string( "MotorDefn::ParseXML" ) );
99  }
100  // type: cantalon, sparkmax_brushless and sparkmax_brushed are valid
101  else if ( strcmp( attr.name(), "type" ) == 0 )
102  {
103  mtype = attr.value();
104  }
105  // inverted
106  else if ( strcmp( attr.name(), "inverted" ) == 0 )
107  {
108  inverted = attr.as_bool();
109  }
110  // sensor inverted
111  else if ( strcmp( attr.name(), "sensorInverted" ) == 0 )
112  {
113  sensorInverted = attr.as_bool();
114  }
115  else if ( strcmp( attr.name(), "feedbackDevice" ) == 0 )
116  {
117  auto val = string( attr.value() );
118  if ( val.compare( "QUADENCODER") == 0 )
119  {
120  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::QuadEncoder;
121  }
122  else if ( val.compare( "ANALOG") == 0 )
123  {
124  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::Analog;
125  }
126  else if ( val.compare( "TACHOMETER") == 0 )
127  {
128  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::Tachometer;
129  }
130  else if ( val.compare( "PULSEWIDTHENCODERPOSITION") == 0 )
131  {
132  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::PulseWidthEncodedPosition;
133  }
134  else if ( val.compare( "SENSORSUM") == 0 )
135  {
136  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::SensorSum;
137  }
138  else if ( val.compare( "SENSORDIFFERENCE") == 0 )
139  {
140  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::SensorDifference;
141  }
142  else if ( val.compare( "REMOTESENSOR0") == 0 )
143  {
144  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::RemoteSensor0;
145  }
146  else if ( val.compare( "REMOTESENSOR1") == 0 )
147  {
148  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::RemoteSensor1;
149  }
150  else if ( val.compare( "SOFTWAREEMULATEDSENSOR") == 0 )
151  {
152  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::SoftwareEmulatedSensor;
153  }
154  else if ( val.compare( "CTRE_MAGENCODER_ABSOLUTE") == 0 )
155  {
156  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::CTRE_MagEncoder_Absolute;
157  }
158  else if ( val.compare( "CTRE_MAGENCODER_RELATIVE") == 0 )
159  {
160  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::CTRE_MagEncoder_Relative;
161  }
162  else if ( val.compare( "INTERNAL") == 0 )
163  {
164  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::IntegratedSensor;
165  }
166  else if ( val.compare( "NONE") == 0 )
167  {
168  feedbackDevice = ctre::phoenix::motorcontrol::FeedbackDevice::None;
169  }
170  else
171  {
172  string msg = "Invalid feedback device ";
173  msg += val;
174  Logger::GetLogger()->LogError( string("MotorDefn::ParseXML "), msg );
175  }
176  }
177  // counts per revolution
178  else if ( strcmp( attr.name(), "countsPerRev" ) == 0 )
179  {
180  countsPerRev = attr.as_int();
181  }
182  // gear ratio
183  else if ( strcmp( attr.name(), "gearRatio" ) == 0 )
184  {
185  gearRatio = attr.as_float();
186  }
187 
188  // brake mode (or coast)
189  else if ( strcmp( attr.name(), "brakeMode" ) == 0 )
190  {
191  brakeMode = attr.as_bool();
192  }
193  // slaveto (existing CAN id of the master motor)
194  else if ( strcmp( attr.name(), "slaveTo") == 0 )
195  {
196  slaveTo = attr.as_int();
197  }
198  // peak current duration (cantalon)
199  else if ( strcmp( attr.name(), "peakCurrentDuration") == 0 )
200  {
201  peakCurrentDuration = attr.as_int();
202  }
203  // continuous current duration (cantalon)
204  else if ( strcmp( attr.name(), "continuousCurrentLimit") == 0 )
205  {
206  continuousCurrentLimit = attr.as_int();
207  }
208  // peak current limit (cantalon)
209  else if ( strcmp( attr.name(), "peakCurrentLimit") == 0 )
210  {
211  peakCurrentLimit = attr.as_int();
212  }
213  // enable current limiting
214  else if ( strcmp( attr.name(), "currentLimiting") == 0 )
215  {
216  enableCurrentLimit = attr.as_bool();
217  }
218  else if ( strcmp( attr.name(), "forwardlimitswitch") == 0 )
219  {
220  forwardLimitSwitch = attr.as_bool();
221  }
222  else if ( strcmp( attr.name(), "forwardlimitswitchopen") == 0 )
223  {
224  forwardLimitSwitchNormallyOpen = attr.as_bool();
225  }
226  else if ( strcmp( attr.name(), "reverselimitswitch") == 0 )
227  {
228  reverseLimitSwitch = attr.as_bool();
229  }
230  else if ( strcmp( attr.name(), "reverselimitswitchopen") == 0 )
231  {
232  reverseLimitSwitchNormallyOpen = attr.as_bool();
233  }
234  else
235  {
236  printf( "==>>MotorDefn::ParseXML invalid attribute %s \n", attr.name() );
237  hasError = true;
238  }
239  }
240 
241  if ( !hasError )
242  {
243  pdpID = ( pdpID < 0 ) ? canID : pdpID;
244  controller = DragonMotorControllerFactory::GetInstance()->CreateMotorController( mtype,
245  canID,
246  pdpID,
247  usage,
248  inverted,
249  sensorInverted,
250  feedbackDevice,
251  countsPerRev,
252  gearRatio,
253  brakeMode,
254  slaveTo,
255  peakCurrentLimit,
256  peakCurrentDuration,
257  peakCurrentLimit,
258  enableCurrentLimit,
259  forwardLimitSwitch,
260  forwardLimitSwitchNormallyOpen,
261  reverseLimitSwitch,
262  reverseLimitSwitchNormallyOpen );
263  }
264  return controller;
265 }

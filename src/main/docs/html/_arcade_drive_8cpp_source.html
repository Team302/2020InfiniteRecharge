133 (
134  double throttle,
135  double steer,
136  double* left,
137  double* right
138 )
139 {
140  // convert throttle / steer values to left / right values
141  *left = (throttle + steer);
142  *right = (throttle - steer);
143 
144  // make sure the values are within -1.0 to 1.0
145  auto maxValue = max(abs( *left ), abs( *right ) );
146  if ( maxValue > 1.0 )

160 void ArcadeDrive::CurvatureDrive
161 (
162  double throttle,
163  double steer,
164  double* left,
165  double* right
166 )
167 {
168  auto isQuickTurn = GetController() != nullptr ? GetController()->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::CURVATURE_DRIVE_QUICK_TURN ) : false;
169 
170  double angularPower = steer;
171 
172  if (isQuickTurn)
173  {
174  if (abs(throttle) < m_quickStopThreshold)
175  {
176  m_quickStopAccumulator = (1 - m_quickStopAlpha) * m_quickStopAccumulator + m_quickStopAlpha * steer * 2;
177  }
178  }
179  else
180  {
181  angularPower = abs(throttle) * steer - m_quickStopAccumulator;
182 
183  if (m_quickStopAccumulator > 1.0)
184  {
185  m_quickStopAccumulator -= 1.0;
186  }
187  else if (m_quickStopAccumulator < -1.0)
188  {
189  m_quickStopAccumulator += 1.0;
190  }
191  else
192  {
193  m_quickStopAccumulator = 0.0;
194  }
195  }
196 
197  *left = throttle + angularPower;
198  *right = throttle - angularPower;
199 
200  // If rotation is overpowered, reduce both outputs to within acceptable range
201  if (isQuickTurn)
202  {
203  if (*left > 1.0)
204  {
205  *right -= *left - 1.0;
206  *left = 1.0;
207  }
208  else if (*right > 1.0)
209  {
210  *left -= *right - 1.0;
211  *right = 1.0;
212  }
213  else if (*left < -1.0)
214  {
215  *right -= *left + 1.0;
216  *left = -1.0;
217  }
218  else if (*right < -1.0)
219  {
220  *left -= *right + 1.0;
221  *right = -1.0;
222  }
223  }
224 
225  // make sure the values are within -1.0 to 1.0
226  auto maxValue = max(abs( *left ), abs( *right ) );
227  if ( maxValue > 1.0 )
228  {
229  *left /= maxValue;
230  *right /= maxValue;
231  }
232 
233 }

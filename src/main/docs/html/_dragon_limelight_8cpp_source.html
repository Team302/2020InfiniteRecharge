106 double DragonLimelight::GetTargetHorizontalOffset() const
107 {
108  double tx = m_networktable.get()->GetNumber("tx", 0.0);
109  double ty = m_networktable.get()->GetNumber("ty", 0.0);
110  if(m_rotation == 0.0)
111  {
112  return tx;
113  }
114  else if(m_rotation == 90.0)
115  {
116  return -ty;
117  }
118  else if(m_rotation == 180.0)
119  {
120  return -tx;
121  }
122  else if(m_rotation == 270.0)
123  {
124  return ty;
125  }
126  else
127  {
128  Logger::GetLogger()->LogError("DragonLimelight::GetTargetVerticalOffset", "Invalid limelight rotation");
129  return -1.0;
130  }

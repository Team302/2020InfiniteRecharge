63 double DragonLimelight::GetDistance() const
64 {
65  double distance = -1.0;
66  if ( HasTarget() )
67  {
68  distance = (m_targetHeight - m_mountHeight) / tan( m_mountingAngle + GetTargetVerticalOffset() );
69  }

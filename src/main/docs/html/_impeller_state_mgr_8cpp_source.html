39 ImpellerStateMgr::ImpellerStateMgr() : m_stateMap(),
40  m_currentState()
41 {
42  // Parse the configuration file
43  auto stateXML = make_unique<StateDataDefn>();
44  vector<MechanismTargetData*> targetData = stateXML.get()->ParseXML( MechanismTypes::MECHANISM_TYPE::IMPELLER );
45 
46  // initialize the xml string to state map
47  map<string, IMPELLER_STATE> stateMap;
48  stateMap["IMPELLEROFF"] = IMPELLER_STATE::OFF;
49  stateMap["IMPELLERON"] = IMPELLER_STATE::ON;
50  stateMap["IMPELLERAGITATE"] = IMPELLER_STATE::AGITATE;
51 
52  // create the states passing the configuration data
53  for ( auto td: targetData )
54  {
55  auto stateString = td->GetStateString();
56  auto stateItr = stateMap.find( stateString );
57  if ( stateItr != stateMap.end() )
58  {
59  auto stateEnum = stateItr->second;
60  auto stateIt = m_stateMap.find( stateEnum );
61  if ( stateIt == m_stateMap.end() )
62  {
63  switch ( stateEnum )
64  {
65  case IMPELLER_STATE::ON:
66  { // todo update the constructor
67  auto thisState = new ImpellerOn();
68  m_stateMap[IMPELLER_STATE::ON] = thisState;
69  m_currentState = thisState;
70  m_currentStateEnum = stateEnum;
71  m_currentState->Init();
72  }
73  break;
74 
75  case IMPELLER_STATE::OFF:
76  { // todo update the constructor
77  auto thisState = new ImpellerOff();
78  m_stateMap[IMPELLER_STATE::OFF] = thisState;
79  }
80  break;
81 
82  case IMPELLER_STATE::AGITATE:
83  {
84 
85  }
86  break;
87 
88  default:
89  {
90  Logger::GetLogger()->LogError( string(""), string(""));
91  }
92  break;
93  }
94  }
95  Logger::GetLogger()->LogError( string(""), string(""));
96  }
97  else
98  {
99  Logger::GetLogger()->LogError( string(""), string(""));
100  }
101  }
102 }

106 void ImpellerStateMgr::RunCurrentState()
107 {
108  // process teleop/manual interrupts
109  auto controller = TeleopControl::GetInstance();
110  if ( controller != nullptr )
111  {
112  if ( controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::IMPELLER_SPIN ) &&
113  m_currentStateEnum != IMPELLER_STATE::ON )
114  {
115  SetCurrentState( IMPELLER_STATE::ON, false );
116  }
117  else if ( controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::IMPELLER_STOP ) &&
118  m_currentStateEnum != IMPELLER_STATE::OFF )
119  {
120  SetCurrentState( IMPELLER_STATE::OFF, false );
121  }
122  else if ( controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::IMPELLER_AGITATE ) &&
123  m_currentStateEnum != IMPELLER_STATE::AGITATE )
124  {
125  SetCurrentState( IMPELLER_STATE::AGITATE, false );
126  }
127  }
128 
129  // run the current state
130  if ( m_currentState != nullptr )
131  {
132  m_currentState->Run();
133  }
134

51 ImpellerStateMgr::ImpellerStateMgr() :
52  m_stateVector(),
53  m_currentState(),
54  m_impeller( MechanismFactory::GetMechanismFactory()->GetIMechanism(MechanismTypes::MECHANISM_TYPE::IMPELLER) ),
55  m_reverse( false ),
56  m_numReverseLoops( 0 ),
57  m_consecZero( 0 )
58 {
59  // Parse the configuration file
60  auto stateXML = make_unique<StateDataDefn>();
61  vector<MechanismTargetData*> targetData = stateXML.get()->ParseXML( MechanismTypes::MECHANISM_TYPE::IMPELLER );
62 
63  // initialize the xml string to state map
64  map<string, IMPELLER_STATE> stateStringToEnumMap;
65  stateStringToEnumMap["IMPELLEROFF"] = IMPELLER_STATE::OFF;
66  stateStringToEnumMap["IMPELLERHOLD"] = IMPELLER_STATE::HOLD;
67  stateStringToEnumMap["IMPELLERAGITATE"] = IMPELLER_STATE::AGITATE;
68  stateStringToEnumMap["IMPELLERTOSHOOTER"] = IMPELLER_STATE::TO_SHOOTER;
69  m_stateVector.resize(4);
70  // create the states passing the configuration data
71  for ( auto td: targetData )
72  {
73  auto stateString = td->GetStateString();
74  auto stateStringToEnumMapItr = stateStringToEnumMap.find( stateString );
75  if ( stateStringToEnumMapItr != stateStringToEnumMap.end() )
76  {
77  auto stateEnum = stateStringToEnumMapItr->second;
78  if ( m_stateVector[stateEnum] == nullptr )
79  {
80  auto controlData = td->GetController();
81  auto target = td->GetTarget();
82  auto solState = td->GetSolenoidState();
83  auto fbControlData = td->GetFailoverController(); // todo pass through to the states
84  auto fbTarget = td->GetFailoverTarget(); // todo pass through to the states
85 
86  switch ( stateEnum )
87  {
88  case IMPELLER_STATE::HOLD:
89  {
90  auto thisState = new ImpellerState(controlData, target, solState );
91  m_stateVector[stateEnum] = thisState;
92  Logger::GetLogger()->LogError("ImpellerStateMgr::ImpellerStateMgr", "Impeller Hold State added to Map");
93  }
94  break;
95 
96  case IMPELLER_STATE::TO_SHOOTER:
97  {
98  auto thisState = new ImpellerState(controlData, target, solState );
99  m_stateVector[stateEnum] = thisState;
100  Logger::GetLogger()->LogError("ImpellerStateMgr::ImpellerStateMgr", "Impeller To Shooter State added to Map");
101  }
102  break;
103 
104  case IMPELLER_STATE::OFF:
105  {
106  auto thisState = new ImpellerState(controlData, target, solState );
107  m_stateVector[stateEnum] = thisState;
108  m_currentState = thisState;
109  m_currentStateEnum = stateEnum;
110  m_currentState->Init();
111  Logger::GetLogger()->LogError("ImpellerStateMgr::ImpellerStateMgr", "Impeller Off State added to Map");
112  }
113  break;
114 
115  case IMPELLER_STATE::AGITATE:
116  {
117  auto thisState = new ImpellerState(controlData, target, solState );
118  m_stateVector[stateEnum] = thisState;
119  Logger::GetLogger()->LogError("ImpellerStateMgr::ImpellerStateMgr", "Impeller Agitate State added to Map");
120  }
121  break;
122 
123  default:
124  {
125  Logger::GetLogger()->LogError( string("ImpellerStateMgr::ImpellerStateMgr"), string("unknown state"));
126  }
127  break;
128  }
129  }
130  else
131  {
132  Logger::GetLogger()->LogError( string("ImpellerStateMgr::ImpellerStateMgr"), string("multiple mechanism state info for state"));
133  }
134  }
135  else
136  {
137  Logger::GetLogger()->LogError( string("ImpellerStateMgr::ImpellerStateMgr"), string("state not found"));
138  }
139  }

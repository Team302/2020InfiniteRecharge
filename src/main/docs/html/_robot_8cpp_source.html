293 void Robot::TestInit()
294 {
295  m_testSelected = m_testChooser.GetSelected();
296  if ( m_testSelected == m_buttonBoxTest )
297  {
298  m_currentTest = BUTTON_BOX;
299  m_buttonBoxDisplay = new ButtonBoxDisplay();
300  }
301  else if ( m_testSelected == m_dragonXBoxTest )
302  {
303  m_currentTest = XBOX;
304  m_xBoxDisplay = new XboxDisplay();
305  }
306  else if ( m_testSelected == m_intakeTest )
307  {
308  m_currentTest = INTAKE;
309  m_intakeStateMgrTest = new IntakeStateMgrTest();
310  }
311  else if ( m_testSelected == m_impellerTest )
312  {
313  m_currentTest = IMPELLER;
314  m_impellerStateMgrTest = new ImpellerStateMgrTest();
315  }
316  else if ( m_testSelected == m_ballTransferTest )
317  {
318  m_currentTest = TRANSFER;
319  m_ballTransferStateMgrTest = new BallTransferStateMgrTest();
320  }
328  else
329  {
330  m_currentTest = NONE;
331  }
332 }

337 void Robot::TestPeriodic()
338 {

341  case BUTTON_BOX:
342  m_buttonBoxDisplay->periodic();
343  break;
344 
345  case XBOX:
346  m_xBoxDisplay->periodic();
347  break;
348 
349  case INTAKE:
350  if ( !m_intakeStateMgrTest->IsDone() )
351  {
352  m_intakeStateMgrTest->Periodic();
353  }
354  break;
355 
356  case IMPELLER:
357  if ( !m_impellerStateMgrTest->IsDone() )
358  {
359  m_impellerStateMgrTest->Periodic();
360  }
361  break;
362 
363  case TRANSFER:
364  if ( !m_ballTransferStateMgrTest->IsDone() )
365  {
366  m_ballTransferStateMgrTest->Periodic();
367  }
368  break;
377  default:
378  break;

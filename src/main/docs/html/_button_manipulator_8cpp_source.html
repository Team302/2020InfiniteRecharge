80 void BallManipulator::SetCurrentState
81 (
82  BALL_MANIPULATOR_STATE state
83 )
84 {
85  m_currentState = state;
86  switch ( state )
87  {
88  case BALL_MANIPULATOR_STATE::OFF:
89  m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
90  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::OFF, false );
91  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::OFF, false );
92  //m_turret->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
93  m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
94  //m_hood->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
95  break;
96 
97  case BALL_MANIPULATOR_STATE::INTAKE:
98  m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::ON, false );
99  if ( m_impeller->GetCurrentState() == ImpellerStateMgr::IMPELLER_STATE::OFF )
100  {
101  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::HOLD, false );
102  }
103  if ( m_transfer->GetCurrentState() == BallTransferStateMgr::BALL_TRANSFER_STATE::OFF )
104  {
105  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_IMPELLER , false );
106  }
107  break;
108 
109  case BALL_MANIPULATOR_STATE::INTAKE_HUMAN_PLAYER:
110  m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::ON, false ); // todo: react to change
111  if ( m_impeller->GetCurrentState() == ImpellerStateMgr::IMPELLER_STATE::OFF )
112  {
113  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::HOLD, false );
114  }
115  if ( m_transfer->GetCurrentState() == BallTransferStateMgr::BALL_TRANSFER_STATE::OFF )
116  {
117  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_IMPELLER , false );
118  }
119  break;
120 
121  case BALL_MANIPULATOR_STATE::HOLD:
122  m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
123  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::HOLD, false );
124  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_IMPELLER , false );
125  //m_turret->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
126  m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::GET_READY, false );
127  //m_hood->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
128  break;
129 
130  case BALL_MANIPULATOR_STATE::GET_READY_TO_SHOOT:
131  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::HOLD, false );
132  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_IMPELLER, false );
133  //m_turret->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
134  m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::GET_READY, false );
135  //m_hood->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
136  break;
137 
138  case BALL_MANIPULATOR_STATE::SHOOT:
139  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::TO_SHOOTER, false );
140  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, false );
141  //m_turret->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
142  m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::SHOOT, false );
143  //m_hood->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
144  break;
145 
146  default:
147  break;
148  }
149  RunCurrentState();
150 
151 }

73 void DragonFalcon::SetControlMode(ControlModes::CONTROL_TYPE mode)
74 {
75  switch (mode)
76  {
77  case ControlModes::CONTROL_TYPE::PERCENT_OUTPUT:
78  case ControlModes::CONTROL_TYPE::VOLTAGE:
79  m_controlMode = ctre::phoenix::motorcontrol::ControlMode::PercentOutput;
80  break;
81 
82  case ControlModes::CONTROL_TYPE::POSITION_DEGREES:
83  case ControlModes::CONTROL_TYPE::POSITION_INCH:
84  m_controlMode = ctre::phoenix::motorcontrol::ControlMode::Position;
85  break;
86 
87  case ControlModes::CONTROL_TYPE::VELOCITY_DEGREES:
88  case ControlModes::CONTROL_TYPE::VELOCITY_INCH:
89  m_controlMode = ctre::phoenix::motorcontrol::ControlMode::Velocity;
90  break;
91 
92  case ControlModes::CONTROL_TYPE::CURRENT:
93  m_controlMode = ctre::phoenix::motorcontrol::ControlMode::Current;
94  break;
95 
96  case ControlModes::CONTROL_TYPE::MOTION_PROFILE:
97  m_controlMode = ctre::phoenix::motorcontrol::ControlMode::MotionProfile;
98  break;
99 
100  case ControlModes::CONTROL_TYPE::MOTION_PROFILE_ARC:
101  m_controlMode = ctre::phoenix::motorcontrol::ControlMode::MotionProfileArc;
102  break;
103 
104  case ControlModes::CONTROL_TYPE::TRAPEZOID:
105  m_controlMode = ctre::phoenix::motorcontrol::ControlMode::MotionMagic;
106  break;
107 
108  default:
109  Logger::GetLogger()->LogError( string("DragonFalcon::SetControlMode"), string("Invalid Control Mode"));
110  m_controlMode = ctre::phoenix::motorcontrol::ControlMode::PercentOutput;
111  break;
112  }
113 }

127 void DragonFalcon::Set(double value)
128 {
129  switch (m_controlMode)
130  {
131  case ctre::phoenix::motorcontrol::ControlMode::PercentOutput:
132  m_talon->Set( m_controlMode, value );
133  break;
134 
135  case ctre::phoenix::motorcontrol::ControlMode::Position:
136  m_talon->Set(m_controlMode, (value * m_countsPerRev / m_gearRatio) + m_tickOffset);
137  break;
138 
139  case ctre::phoenix::motorcontrol::ControlMode::Velocity:
140  m_talon->Set(m_controlMode, value * m_gearRatio / 600.0);
141  break;
142 
143  case ctre::phoenix::motorcontrol::ControlMode::Current:
144  m_talon->Set(m_controlMode, value );
145  break;
146 
147  case ctre::phoenix::motorcontrol::ControlMode::MotionProfile:
148  m_talon->Set(m_controlMode, value );
149  break;
150 
151  case ctre::phoenix::motorcontrol::ControlMode::MotionProfileArc:
152  m_talon->Set(m_controlMode, value );
153  break;
154 
155  case ctre::phoenix::motorcontrol::ControlMode::MotionMagic:
156  m_talon->Set(m_controlMode, (value * m_countsPerRev / m_gearRatio) + m_tickOffset);
157  break;
158 
159  default:
160  m_talon->Set(ctre::phoenix::motorcontrol::ControlMode::PercentOutput, 0);
161  break;
162  }
163 }

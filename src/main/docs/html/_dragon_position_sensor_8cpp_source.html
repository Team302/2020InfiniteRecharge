77 double DragonPositionSensor::GetDistance()
78 {
79  double dist = 0.0;
80  if ( m_angleSensor != nullptr )
81  {
82  auto currAngle = m_angleSensor->GetRawAngle();
83  auto deltaAngle = currAngle - m_startAngle;
84  auto deltaDist = deltaAngle * M_PI * m_diameter;
85  dist = m_startDist + deltaDist;
86  }

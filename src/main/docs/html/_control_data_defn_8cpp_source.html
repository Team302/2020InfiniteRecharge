81  // parse/validate xml
82  for (pugi::xml_attribute attr = PIDNode.first_attribute(); attr; attr = attr.next_attribute())
83  {
84  if ( strcmp( attr.name(), "identifier") == 0 )
85  {
86  identifier = string( attr.value() );
87  }
88  else if ( strcmp( attr.name(), "mode" ) == 0 )
89  {
90  auto it = modeMap.find( string(attr.value() ) );
91  if ( it != modeMap.end() )
92  {
93  mode = it->second;
94  }
95  }
96  else if (( attr.name(), "constrolServer" ) == 0 )
97  {
98  auto itr = serverMap.find( string( attr.value() ) );
99  if ( itr != serverMap.end() )
100  {
101  server = itr->second;
102  }
103  }
104  else if ( strcmp( attr.name(), "proportional") == 0 )
105  {
106  p = attr.as_double();
107  }
108  else if ( strcmp( attr.name(), "integral") == 0 )
109  {
110  i = attr.as_double();
111  }
112  else if ( strcmp( attr.name(), "derivative") == 0 )
113  {
114  d = attr.as_double();
115  }
116  else if ( strcmp( attr.name(), "feedforward") == 0 )
117  {
118  f = attr.as_double();
119  }
120  else if ( strcmp( attr.name(), "izone") == 0 )
121  {
122  izone = attr.as_double();
123  }
124  else if ( strcmp( attr.name(), "maxacceleration") == 0 )
125  {
126  maxAccel = attr.as_double();
127  }
128  else if ( strcmp( attr.name(), "cruisevelocity") == 0 )
129  {
130  cruiseVel = attr.as_double();
131  }
132  else if ( strcmp( attr.name(), "peak") == 0 )
133  {
134  peak = attr.as_double();
135  }
136  else if ( strcmp( attr.name(), "nominal") == 0 )
137  {
138  nominal = attr.as_double();
139  }
140  else
141  {
142  printf( "==>> ControlDataDefn::ParseXML invalid attribute %s \n", attr.name() );
143  hasError = true;
144  }
145  }
146  if ( !hasError )
147  {
148  data = new ControlData( mode, server, identifier, p, i, d, f, izone, maxAccel, cruiseVel, peak, nominal );
149  }

151 }

50 double GTADrive::GetThrottle()
51 {
52  auto controller = GetController();
53  double throttle = 0.0;
54  if ( controller != nullptr )
55  {
56  auto forward = controller->GetAxisValue( TeleopControl::FUNCTION_IDENTIFIER::GTA_DRIVE_FORWARD);
57  auto backward = controller->GetAxisValue( TeleopControl::FUNCTION_IDENTIFIER::GTA_DRIVE_BACKWARD);
58  throttle = forward - backward;
59  }
60  return throttle;
61 }

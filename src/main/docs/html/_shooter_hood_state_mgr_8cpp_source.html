45

48  stateMap["MOVEUP"] = SHOOTER_HOOD_STATE::MOVE_UP;

51  stateMap["MANUAL"] = SHOOTER_HOOD_STATE::MANUAL;
52 
53  // create the states passing the configuration data
54  for ( auto td: targetData )
55  {
56  auto stateString = td->GetStateString();
57  auto stateItr = stateMap.find( stateString );
58  if ( stateItr != stateMap.end() )
59  {
60  auto stateEnum = stateItr->second;
61  auto stateIt = m_stateMap.find( stateEnum );
62  if ( stateIt == m_stateMap.end() )
63  {
64  auto controlData = td->GetController();
65  auto target = td->GetTarget();
66 
67  switch ( stateEnum )
68  {
69  case SHOOTER_HOOD_STATE::MOVE_UP:
70  { // todo update the constructor take in controlData and target
71  auto thisState = new ShooterHoodMoveUp(controlData, target, MechanismTargetData::SOLENOID::NONE);
72  m_stateMap[SHOOTER_HOOD_STATE::MOVE_UP] = thisState;
73  }
74  break;
75 
76  case SHOOTER_HOOD_STATE::MOVE_DOWN:
77  { // todo update the constructor take in controlData and target
78  auto thisState = new ShooterHoodMoveDown(controlData, target, MechanismTargetData::SOLENOID::NONE);
79  m_stateMap[SHOOTER_HOOD_STATE::MOVE_DOWN] = thisState;
80  }
81  break;
82 
83  case SHOOTER_HOOD_STATE::HOLD_POSITION:
84  { // todo update the constructor take in controlData and target
85  auto thisState = new ShooterHoodHoldPosition(controlData, target, MechanismTargetData::SOLENOID::NONE);
86  m_stateMap[SHOOTER_HOOD_STATE::HOLD_POSITION] = thisState;

92

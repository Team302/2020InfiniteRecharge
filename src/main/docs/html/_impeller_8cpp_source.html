129 double Impeller::GetCurrentSpeed() const
130 {
131  double speed = 0.0;
132  if ( m_motor.get() != nullptr && m_motor.get()->GetCurrent() > 10.0 )
133  {
134  speed = 0.0;
135  }
136  else if ( m_encoder.get() != nullptr )
137  {
138  auto vel = m_encoder.get()->GetVelocity();
139  auto time = m_encoder.get()->GetLastTimestamp();
140  if ( time > m_lastTimeStamp )
141  {
142  speed = vel*10.0;
143  m_lastTimeStamp = time;
144  m_lastVelocity = speed;
145  }
146  }
147  return speed;
148 }

300 //------------------------------------------------------------------
301 // Method: SetAxisScaleFactor
302 // Description: Allow the range of values to be set smaller than
303 // -1.0 to 1.0. By providing a scale factor between 0.0
304 // and 1.0, the range can be made smaller. If a value
305 // outside the range is provided, then the value will
306 // be set to the closest bounding value (e.g. 1.5 will
307 // become 1.0)
308 // Returns: void
309 //------------------------------------------------------------------
310 void TeleopControl::SetAxisScaleFactor
311 (
312  TeleopControl::FUNCTION_IDENTIFIER function, // <I> - function that will update an axis
313  float scaleFactor // <I> - scale factor used to limit the range
314 )
315 {
316  int ctlIndex = m_controllerIndex[ function];
317  IDragonGamePad::AXIS_IDENTIFIER axis = m_axisIDs[ function ];
318  if ( ctlIndex > -1 && axis != IDragonGamePad::AXIS_IDENTIFIER::UNDEFINED_AXIS )
319  {
320  if (m_controllers[ ctlIndex ] != nullptr)
321  {
322  m_controllers[ ctlIndex ]->SetAxisScale( axis,scaleFactor);
323  }
324  }
325 }

354 //------------------------------------------------------------------

38 MechanismTargetData* MechanismTargetDefn::ParseXML
39 (
40  xml_node MechanismDataNode
41 )
42 {
43  // initialize output
44  MechanismTargetData* mechData = nullptr;
45 
46  bool hasError = false;
47 
48  string stateName;
49  string controllerIdentifier;
50  double target = 0.0;
51  MechanismTargetData::SOLENOID solenoid = MechanismTargetData::SOLENOID::NONE;
52  string failoverControllerIdentifier;
53  double failoverTarget = 0.0;
54 
55  // parse/validate xml
56  for (xml_attribute attr = MechanismDataNode.first_attribute(); attr; attr = attr.next_attribute())
57  {
58  if ( strcmp( attr.name(), "stateIdentifier" ) == 0 )
59  {
60  stateName = string( attr.value() );
61  }
62  else if ( strcmp( attr.name(), "controlDataIdentifier" ) == 0 )
63  {
64  controllerIdentifier = string( attr.value() );
65  }
66  else if ( strcmp( attr.name(), "value") == 0 )
67  {
68  target = attr.as_double();
69  }
70  else if( strcmp( attr.name(), "solenoid" ) == 0 )
71  {
72  auto val = attr.value();
73  if(strcmp(val, "ON") == 0)
74  {
75  solenoid = MechanismTargetData::SOLENOID::ON;
76  }
77  else if ( strcmp(val, "OFF") == 0)
78  {
79  solenoid = MechanismTargetData::SOLENOID::OFF;
80  }
81  else if( strcmp(val, "NONE") == 0)
82  {
83  solenoid = MechanismTargetData::SOLENOID::NONE;
84  }
85  else
86  {
87  Logger::GetLogger()->LogError( string("MechanismTargetDefn::ParseXML"), string("solenoid enum"));
88  }
89  }
90  else if ( strcmp( attr.name(), "failoverControlDataIdentifier") == 0 )
91  {
92  failoverControllerIdentifier = string( attr.value() );
93  }
94  else if ( strcmp( attr.name(), "failoverValue") == 0 )
95  {
96  failoverTarget = attr.as_double();
97  }
98  else
99  {
100  string msg = "unknown attribute ";
101  msg += attr.name();
102  Logger::GetLogger()->LogError( string("MechanismTargetDefn::ParseXML"), msg );
103  hasError = true;
104  }
105  }
106 
107  if ( !hasError && !stateName.empty() && !controllerIdentifier.empty() )
108  {
109  mechData = new MechanismTargetData( stateName, controllerIdentifier, target, solenoid, failoverControllerIdentifier, failoverTarget );
110  }
111  else
112  {
113  Logger::GetLogger()->LogError( string("MechanismTargetDefn::ParseXML"), string("incomplete date"));
114  }
115 
116  return mechData;
117 }

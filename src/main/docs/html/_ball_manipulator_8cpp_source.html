67 void BallManipulator::RunCurrentState()
68 {
69  auto controller = TeleopControl::GetInstance();
70 
71  // process teleop/manual interrupts
72 
73  if ( controller != nullptr )
74  {
75  if (controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::OFF))
76  {
77  SetCurrentState( BALL_MANIPULATOR_STATE::OFF);
78  }
79  if (controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::INTAKE))
80  {
81  SetCurrentState(BALL_MANIPULATOR_STATE::INTAKE);
82  }
83  if (controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::INTAKE_HUMAN_PLAYER))
84  {
85  SetCurrentState( BALL_MANIPULATOR_STATE::INTAKE_HUMAN_PLAYER);
86  }
87  if (controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::GET_READY_TO_SHOOT))
88  {
89  SetCurrentState(BALL_MANIPULATOR_STATE ::GET_READY_TO_SHOOT);
90  }
91  if (controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::SHOOT))
92  {
93  SetCurrentState(BALL_MANIPULATOR_STATE::SHOOT);
94  }
95  if (controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::UNJAM_CLOCKWISE))
96  {
97  SetCurrentState(BALL_MANIPULATOR_STATE::UNJAM_CLOCKWISE);
98  }
99  if (controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::UNJAM_COUNTERCLOCKWISE))
100  {
101  SetCurrentState(BALL_MANIPULATOR_STATE::UNJAM_COUNTERCLOCKWISE);
102  }
103  if (controller->IsButtonPressed( TeleopControl::FUNCTION_IDENTIFIER::TRANSFER_UP))
104  {
105  SetCurrentState(BALL_MANIPULATOR_STATE::TRANSFER_UP);
106  }
107  if (controller->IsButtonPressed (TeleopControl::FUNCTION_IDENTIFIER::TRANSFER_DOWN))
108  {
109  SetCurrentState(BALL_MANIPULATOR_STATE::TRANSFER_DOWN);
110  }
111 
112  }
113 
114  //SetCurrentState(m_currentState);
115  // run the current state
116  m_intake->RunCurrentState();
117  m_impeller->RunCurrentState();
118  m_transfer->RunCurrentState();
119  m_turret->RunCurrentState();
120  m_shooter->RunCurrentState();
121  m_hood->RunCurrentState();
122 
123 }

127 void BallManipulator::SetCurrentState
128 (
129  BALL_MANIPULATOR_STATE state
130 )
131 {
132  m_currentState = state;
133  switch ( state )
134  {
135  case BALL_MANIPULATOR_STATE::OFF:
136  m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
137  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::OFF, false );
138  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::OFF, false );
139  m_turret->SetCurrentState( TurretStateMgr::TURRET_STATE::HOLD, false );
140  m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
141  m_hood->SetCurrentState( ShooterHoodStateMgr::SHOOTER_HOOD_STATE::HOLD_POSITION, false );
142  break;
143 
144  case BALL_MANIPULATOR_STATE::INTAKE:
145  m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::ON, false );
146  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::HOLD, false );
147  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_IMPELLER , false );
148  break;
149 
150  case BALL_MANIPULATOR_STATE::INTAKE_HUMAN_PLAYER:
151  m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::HUMANPLAYER, false );
152  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::HOLD, false );
153  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_IMPELLER , false );
154  break;
155 
156  case BALL_MANIPULATOR_STATE::HOLD:
157  m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
158  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::OFF, false );
159  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_IMPELLER , false );
160  m_turret->SetCurrentState( TurretStateMgr::TURRET_STATE::HOLD, false );
161  m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
162  m_hood->SetCurrentState( ShooterHoodStateMgr::SHOOTER_HOOD_STATE::HOLD_POSITION, false ); // todo auto aim is needed
163  break;
164 
165  case BALL_MANIPULATOR_STATE::GET_READY_TO_SHOOT:
166  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::OFF, false );
167  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_IMPELLER, false );
168  m_turret->SetCurrentState( TurretStateMgr::TURRET_STATE::HOLD, false );
169  //m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
170  m_hood->SetCurrentState( ShooterHoodStateMgr::SHOOTER_HOOD_STATE::HOLD_POSITION, false ); // todo auto aim is needed
171  break;
172 
173  case BALL_MANIPULATOR_STATE::SHOOT:
174  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::TO_SHOOTER, false );
175  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, false );
176  m_turret->SetCurrentState( TurretStateMgr::TURRET_STATE::HOLD, false );
177  //m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
178  m_hood->SetCurrentState( ShooterHoodStateMgr::SHOOTER_HOOD_STATE::HOLD_POSITION, false ); // todo auto aim is needed
179  break;
180 
181  case BALL_MANIPULATOR_STATE::UNJAM_CLOCKWISE:
182  //m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
183  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::TO_SHOOTER, false );
184  //m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_IMPELLER, false );
185  //m_turret->SetCurrentState( TurretStateMgr::TURRET_STATE::HOLD, false );
186  //m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
187  //m_hood->SetCurrentState( ShooterHoodStateMgr::SHOOTER_HOOD_STATE::HOLD_POSITION, false );
188  break;
189 
190  case BALL_MANIPULATOR_STATE::UNJAM_COUNTERCLOCKWISE:
191  //m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
192  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::HOLD, false );
193  //m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_IMPELLER, false );
194  //m_turret->SetCurrentState( TurretStateMgr::TURRET_STATE::HOLD, false );
195  //m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
196  //m_hood->SetCurrentState( ShooterHoodStateMgr::SHOOTER_HOOD_STATE::HOLD_POSITION, false );
197  break;
198 
199  case BALL_MANIPULATOR_STATE::SHOOT_AUTON:
200  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::TO_SHOOTER, false );
201  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, false );
202  m_turret->SetCurrentState( TurretStateMgr::TURRET_STATE::HOLD, false );
203  m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::SHOOT, false );
204  m_hood->SetCurrentState( ShooterHoodStateMgr::SHOOTER_HOOD_STATE::HOLD_POSITION, false );
205  break;// todo auto aim is needed
206 
207  case BALL_MANIPULATOR_STATE::TRANSFER_UP:
208  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::TO_SHOOTER, false);
209  break;
210 
211  case BALL_MANIPULATOR_STATE::TRANSFER_DOWN:
212  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::EJECT, false);
213  break;
214 
215  case BALL_MANIPULATOR_STATE::AUTO_AIM:
216  m_intake->SetCurrentState( IntakeStateMgr::INTAKE_STATE::OFF, false );
217  m_impeller->SetCurrentState( ImpellerStateMgr::IMPELLER_STATE::OFF, false );
218  m_transfer->SetCurrentState( BallTransferStateMgr::BALL_TRANSFER_STATE::OFF, false );
219  m_turret->SetCurrentState( TurretStateMgr::TURRET_STATE::LIMELIGHT_AIM, false );
220  m_shooter->SetCurrentState( ShooterStateMgr::SHOOTER_STATE::OFF, false );
221  m_hood->SetCurrentState( ShooterHoodStateMgr::SHOOTER_HOOD_STATE::HOLD_POSITION, false );
222 
223  default:
224  break;
225  }
226  RunCurrentState();
227 
228 }

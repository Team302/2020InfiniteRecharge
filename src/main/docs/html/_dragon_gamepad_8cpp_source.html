97 double DragonGamepad::GetAxisValue
98 (
99  AXIS_IDENTIFIER axis
100 ) const
101 {
102  double output = 0.0;
103  if ( axis < MAX_GAMEPAD_AXIS )
104  {
105  switch ( axis )//switch method to find out the axes for each gamepad buttons that use axis.
106  {
107  case GAMEPAD_AXIS_16:
108  output = m_gamepad->GetY( frc::GenericHID::kLeftHand ) * m_axisInversionFactor[ axis ];
109  break;
110 
111  case GAMEPAD_AXIS_17:
112  output = m_gamepad->GetY( frc::GenericHID::kRightHand ) * m_axisInversionFactor[ axis ];
113  break;
114 
115  // jw - changed 2 to axis
116  case GAMEPAD_DIAL_22: //Dial has six settings, 22-27.
117  output = m_gamepad->GetY( frc::GenericHID::kLeftHand ) * m_axisInversionFactor[ axis ];
118  break;
119 
120  case GAMEPAD_DIAL_23:
121  output = m_gamepad->GetY( frc::GenericHID::kLeftHand ) * m_axisInversionFactor[ axis ];
122  break;
123 
124  case GAMEPAD_DIAL_24:
125  output = m_gamepad->GetY( frc::GenericHID::kLeftHand ) * m_axisInversionFactor[ axis ];
126  break;
127 
128  case GAMEPAD_DIAL_25:
129  output = m_gamepad->GetY( frc::GenericHID::kLeftHand ) * m_axisInversionFactor[ axis ];
130  break;
131 
132  case GAMEPAD_DIAL_26:
133  output = m_gamepad->GetY( frc::GenericHID::kLeftHand ) * m_axisInversionFactor[ axis ];
134  break;
135 
136  case GAMEPAD_DIAL_27:
137  output = m_gamepad->GetY( frc::GenericHID::kLeftHand ) * m_axisInversionFactor[ axis ];
138  break;
139 
140  default:
141  break;
142 
143  }
144  if (std::abs(output) < APPLY_STANDARD_DEADBAND)
145  {
146  output = 0.0;
147  }
148  else if (std::abs(output)> APPLY_STANDARD_DEADBAND)
149  {
150  if ( m_axisProfile[axis] == CUBED )
151  {
152  output = pow( output, 3.0 );
153  }
154  output = output * m_axisScale[axis];
155  }
156  }
157  return output;
158 }

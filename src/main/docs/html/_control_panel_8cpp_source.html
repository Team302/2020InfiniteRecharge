161 ControlPanelColors::COLOR ControlPanel::GetColorSeen()
162 {
163  frc::Color detectedColor = m_color->GetColor();
164  uint32_t detectedProximity = m_color->GetProximity();
165  std::string colorString;
166  double confidence = 0.0;
167  frc::Color matchedColor = m_colorMatcher->MatchClosestColor(detectedColor, confidence);
168  if(matchedColor == kGreenTarget && confidence >= 0.94 )
169  {
170  return ControlPanelColors::COLOR::BLUE;
171  }
172  else if(matchedColor == kBlueTarget && confidence >= 0.94 )
173  {
174  return ControlPanelColors::COLOR::GREEN;
175  }
176  else if(matchedColor == kYellowTarget && confidence >= 0.9 )
177  {
178  return ControlPanelColors::COLOR::YELLOW;
179  }
180  else if(matchedColor == kBlueTarget && confidence >= 0.9 )
181  {
182  return ControlPanelColors::COLOR::RED;
183  }
184  return ControlPanelColors::COLOR::UNKNOWN;

116 void DragonSparkMax::Set(double value)
117 {
118  auto output = value;
119  switch (m_controlMode)
120  {
121  case ControlModes::CONTROL_TYPE::PERCENT_OUTPUT:
122  m_spark->Set(value);
123  break;
124 
125  case ControlModes::CONTROL_TYPE::POSITION_DEGREES:
126  output = (ConversionUtils::DegreesToCounts(value,m_countsPerRev) / m_gearRatio) + m_outputRotationOffset;
127  m_spark->GetPIDController().SetReference( output, rev::ControlType::kPosition, 0);
128  break;
129  case ControlModes::CONTROL_TYPE::POSITION_INCH:
130  output = (ConversionUtils::InchesToCounts(value, m_countsPerRev, m_diameter) / m_gearRatio) + m_outputRotationOffset;
131  m_spark->GetPIDController().SetReference( output, rev::ControlType::kPosition, 0);
132  break;
133 
134  case ControlModes::CONTROL_TYPE::VELOCITY_DEGREES:
135  output = (ConversionUtils::DegreesPerSecondToCounts100ms( value, m_countsPerRev ) / m_gearRatio) + m_outputRotationOffset;
136  m_spark->GetPIDController().SetReference( output/60.0, rev::ControlType::kVelocity, 0);
137  break;
138 
139  case ControlModes::CONTROL_TYPE::VELOCITY_INCH:
140  output = (ConversionUtils::InchesPerSecondToCounts100ms( value, m_countsPerRev, m_diameter ) / m_gearRatio) + m_outputRotationOffset;
141  m_spark->GetPIDController().SetReference( output/60.0, rev::ControlType::kVelocity, 0);
142  break;
143 
144  case ControlModes::CONTROL_TYPE::VELOCITY_RPS:
145  output = (ConversionUtils::RPSToCounts100ms( value, m_countsPerRev ) / m_gearRatio) + m_outputRotationOffset;
146  m_spark->GetPIDController().SetReference( output/60.0, rev::ControlType::kVelocity, 0);
147  break;
148 
149  case ControlModes::CONTROL_TYPE::CURRENT:
150  output = value;
151  m_spark->GetPIDController().SetReference( output, rev::ControlType::kCurrent, 0);
152  break;
153 
154  case ControlModes::CONTROL_TYPE::MOTION_PROFILE:
155  output = value;
156  break;
157 
158  case ControlModes::CONTROL_TYPE::MOTION_PROFILE_ARC:
159  output = value;
160  break;
161 
162  case ControlModes::CONTROL_TYPE::TRAPEZOID:
163  output = (ConversionUtils::InchesToCounts(value, m_countsPerRev, m_diameter) / m_gearRatio) + m_outputRotationOffset;
164  m_spark->GetPIDController().SetReference( output, rev::ControlType::kSmartMotion, 0);
165  break;
166 
167  case ControlModes::CONTROL_TYPE::VOLTAGE:
168  output = (ConversionUtils::InchesToCounts(value, m_countsPerRev, m_diameter) / m_gearRatio) + m_outputRotationOffset;
169  m_spark->GetPIDController().SetReference( value, rev::ControlType::kVoltage, 0);
170  break;
171 
172  default:
173  Logger::GetLogger()->LogError( string("DragonSparkMax::SetControlMode"), string("Invalid Control Mode"));
174  m_spark->Set(value);

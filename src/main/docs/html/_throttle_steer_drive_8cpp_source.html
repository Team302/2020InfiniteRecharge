53 void ThrottleSteerDrive::Run( )
54 {
55  // Get throttle and steer values from the subclasses
56  auto throttle = GetThrottle();
57  auto steer = GetSteer();
58 
59  // convert throttle / steer values to left / right values
60  auto left = throttle + steer;
61  auto right = throttle - steer;
62 
63  // make sure the values are within -1.0 to 1.0
64  auto maxValue = abs( left );
65  if ( abs(right) > maxValue )
66  {
67  maxValue = abs( right );
68  }
69  if ( maxValue > 1.0 )
70  {
71  left /= maxValue;
72  right /= maxValue;
73  }
74 
75  // Set the percentages
76  m_chassis.get()->SetOutput( ControlModes::PERCENT_OUTPUT, left, right );
77 }

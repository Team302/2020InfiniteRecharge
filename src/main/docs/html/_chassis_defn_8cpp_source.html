55 shared_ptr<IChassis> ChassisDefn::ParseXML
56 (
57  xml_node chassisNode
58 )
59 {
60  shared_ptr<IChassis> chassis;
61  // initialize the attributes to the default values
62  ChassisFactory::CHASSIS_TYPE type = ChassisFactory::CHASSIS_TYPE::TANK_CHASSIS;
63  double wheelDiameter = 0.0;
64  double wheelBase = 0.0;
65  double track = 0.0;
66  bool hasError = false;
67 
68  // process attributes
69  for (xml_attribute attr = chassisNode.first_attribute(); attr && !hasError; attr = attr.next_attribute())
70  {
71  if ( strcmp( attr.name(), "type" ) == 0 )
72  {
73  auto val = string( attr.value() );
74  if ( val.compare( "MECANUM") == 0 )
75  {
76  type = ChassisFactory::CHASSIS_TYPE::MECANUM_CHASSIS;
77  }
78  else if ( val.compare( "TANK" ) == 0 )
79  {
80  type = ChassisFactory::CHASSIS_TYPE::TANK_CHASSIS;
81  }
82  else
83  {
84  string msg = "Unknown Chassis Type";
85  msg += attr.value();
86  Logger::GetLogger()->LogError( string( "ChassisDefn::ParseXML" ), msg );
87  }
88  }
89  else if ( strcmp( attr.name(), "wheelDiameter" ) == 0 )
90  {
91  wheelDiameter = attr.as_double();
92  }
93  else if ( strcmp( attr.name(), "wheelBase" ) == 0 )
94  {
95  wheelBase = attr.as_double();
96  }
97  else if ( strcmp( attr.name(), "track" ) == 0 )
98  {
99  track = attr.as_double();
100  }
101  else // log errors
102  {
103  string msg = "unknown attribute ";
104  msg += attr.name();
105  Logger::GetLogger()->LogError( string("ChassisDefn::ParseXML"), msg );
106  hasError = true;
107  }
108  }
109 
110 
111  // Process child element nodes
112  IDragonMotorControllerMap motors;
113  unique_ptr<MotorDefn> motorXML = make_unique<MotorDefn>();
114 
115  for (xml_node child = chassisNode.first_child(); child; child = child.next_sibling())
116  {
117  if ( strcmp( child.name(), "motor") == 0 )
118  {
119  auto motor = motorXML.get()->ParseXML(child);
120  if ( motor.get() != nullptr )
121  {
122  motors[ motor.get()->GetType() ] = motor ;
123  }
124  }
125  else // log errors
126  {
127  string msg = "unknown child ";
128  msg += child.name();
129  Logger::GetLogger()->LogError( string("ChassisDefn::ParseXML"), msg );
130  }
131  }
132 
133 
134  // create chassis instance
135  if ( !hasError )
136  {
137  auto factory = ChassisFactory::GetChassisFactory();
138  if ( factory != nullptr )
139  {
140  chassis = factory->CreateChassis( type, wheelDiameter, wheelBase, track, motors );
141  }
142  else // log errors
143  {
144  Logger::GetLogger()->LogError( string("ChassisDefn::ParseXML"), string("unable to create chassis") );
145  }
146  }
147  return chassis;
148 }

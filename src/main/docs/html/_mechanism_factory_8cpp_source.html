110 IMechanism* MechanismFactory::CreateIMechanism
111 (
112  MechanismTypes::MECHANISM_TYPE type,
113  const IDragonMotorControllerMap& motorControllers, // <I> - Motor Controllers
114  const DragonSolenoidMap& solenoids,
115  const ServoMap& servos,
116  const DigitalInputMap& digitalInputs,
117  const AnalogInputMap& analogInputs,
118  ColorSensorV3* colorSensor,
119  shared_ptr<CANCoder> canCoder
120 )
121 {
122  // See if the mechanism was created already, if it wasn't create it
123  IMechanism* subsys = m_mechanisms[type];
124  if ( subsys != nullptr )
125  {
126  string msg = "mechanism already exists";
127  msg += to_string( type );
128  Logger::GetLogger()->LogError( string("MechansimFactory::CreateIMechanism" ), msg );
129  }
130  else
131  {
132  // Create the mechanism
133  switch ( type )
134  {
135  case MechanismTypes::MECHANISM_TYPE::INTAKE:
136  {
137  auto motor = GetMotorController( motorControllers, MotorControllerUsage::MOTOR_CONTROLLER_USAGE::INTAKE );
138  if ( motor.get() != nullptr )
139  {
140  auto solenoid = GetSolenoid( solenoids, SolenoidUsage::SOLENOID_USAGE::INTAKE );
141  if ( solenoid.get() != nullptr )
142  {
143  auto intake = new Intake( motor, solenoid );
144  subsys = dynamic_cast<IMechanism*>( intake );
145  }
146  }
147  }
148  break;
149 
150  case MechanismTypes::MECHANISM_TYPE::IMPELLER:
151  {
152  auto motor = GetMotorController( motorControllers, MotorControllerUsage::MOTOR_CONTROLLER_USAGE::IMPELLER );
153  if ( motor.get() != nullptr )
154  {
155  auto impeller = new Impeller(motor, canCoder);
156  subsys = dynamic_cast<IMechanism*>(impeller);
157  }
158  }
159  break;
160 
161  case MechanismTypes::MECHANISM_TYPE::BALL_TRANSFER:
162  {
163  auto motor = GetMotorController( motorControllers, MotorControllerUsage::MOTOR_CONTROLLER_USAGE::BALL_TRANSFER );
164  if ( motor.get() != nullptr )
165  {
166  auto ballTrans = new BallTransfer( motor );
167  subsys = dynamic_cast<IMechanism*>( ballTrans );
168  }
169  }
170  break;
171 
172  case MechanismTypes::MECHANISM_TYPE::SHOOTER:
173  {
174  auto motor1 = GetMotorController( motorControllers, MotorControllerUsage::MOTOR_CONTROLLER_USAGE::SHOOTER_1 );
175  auto motor2 = GetMotorController( motorControllers, MotorControllerUsage::MOTOR_CONTROLLER_USAGE::SHOOTER_2 );
176  if ( motor1.get() != nullptr && motor2.get() != nullptr )
177  {
178  auto shooter = new Shooter(motor1, motor2);
179  subsys = dynamic_cast<IMechanism*>(shooter);
180  }
181  }
182  break;
183 
184  case MechanismTypes::MECHANISM_TYPE::SHOOTER_HOOD:
185  {
186  auto shooterHoodMotor = GetMotorController ( motorControllers, MotorControllerUsage::MOTOR_CONTROLLER_USAGE::SHOOTER_HOOD );
187  if ( shooterHoodMotor.get() != nullptr )
188  {
189  auto shooterHood = new ShooterHood( shooterHoodMotor, canCoder);
190  subsys = dynamic_cast<IMechanism*>(shooterHood);
191  }
192  }
193  break;
194 
195  case MechanismTypes::MECHANISM_TYPE::CONTROL_TABLE_MANIPULATOR:
196  {
197  auto motor = GetMotorController( motorControllers, MotorControllerUsage::MOTOR_CONTROLLER_USAGE::CONTROL_TABLE_MANIPULATOR );
198  if ( motor.get() != nullptr )
199  {
200  auto solenoid = GetSolenoid( solenoids, SolenoidUsage::SOLENOID_USAGE::CONTROL_TABLE_MANIPULATOR );
201  if ( solenoid.get() != nullptr )
202  {
203  if ( colorSensor !=nullptr )
204  {
205  //todo color sensor pointer needs to be added to the constructor
206  auto controlPanel = new ControlPanel( motor, solenoid, colorSensor );
207  subsys = dynamic_cast<IMechanism*>(controlPanel);
208  }
209  }
210  }
211  }
212  break;
213 
214  case MechanismTypes::MECHANISM_TYPE::CLIMBER:
215  {
216  }
217  break;
218 
219  case MechanismTypes::MECHANISM_TYPE::CRAWLER:
220  {
221  }
222  break;
223 
224  case MechanismTypes::MECHANISM_TYPE::TURRET:
225  {
226  auto motor = GetMotorController(motorControllers, MotorControllerUsage::MOTOR_CONTROLLER_USAGE::TURRET);
227  if(motor.get() != nullptr)
228  {
229  auto turret = new Turret(motor);
230  subsys = dynamic_cast<IMechanism*>(turret);
231  }
232  }
233  break;
234 
235  case MechanismTypes::MECHANISM_TYPE::HOOK_DELIVERY:
236  {
237  auto motor = GetMotorController(motorControllers, MotorControllerUsage::MOTOR_CONTROLLER_USAGE::HOOK_DELIVERY);
238  if(motor.get() != nullptr)
239  {
240  auto hook = new HookDelivery(motor);
241  subsys = dynamic_cast<IMechanism*>(hook);
242  }
243  }
244  break;
245 
246  default:
247  {
248  string msg = "unknown Mechanism type ";
249  msg += to_string( type );
250  Logger::GetLogger()->LogError( "MechanismFactory::CreateIMechanism", msg );
251  }
252  break;
253  }
254  m_mechanisms[type] = subsys;
255  }
256 
257  return subsys;
258 }
